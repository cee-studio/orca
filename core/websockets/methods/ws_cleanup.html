ws_cleanup - cleanup a WebSockets handle#

void ws_cleanup(struct websockets *ws)#

Free a WebSockets handle created with ws_init()

Parameters:

  • ws – the WebSockets handle created with ws_init()

Example#

CURLM *mhandle = curl_multi_init();
struct websockets *ws = ws_init(&cbs, mhandle, NULL);
ws_set_url(ws, "wss://example.com", NULL);

/* run the event-loop */
ws_start(ws);

uint64_t tstamp;
bool is_running;
do {
  is_running = ws_easy_run(ws, 5, &tstamp);
} while (is_running);

ws_end(ws);

ws_cleanup(ws);
curl_multi_cleanup(mhandle);
